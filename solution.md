## Return Negative

```js
if (num < 0) {
  return num
} else if (num == 0) {
  return num
} else {
  return num * -1
}
```

## Sum of Positive

```js
let sum = 0
for (let i = 0; i < arr.length; i++) {
  if (arr[i] > 0) {
    sum += arr[i]
  }
}
return sum
```

## Sum Arrays

```js
let sum = 0
if (numbers === 0) {
  return 0
} else {
  for (let i = 0; i < numbers.length; i++) {
    sum += numbers[i]
  }
  return sum
}
```

## Reversed Strings

```js
let string = ''
for (let i = str.length - 1; i >= 0; i--) {
  string += str[i]
}
return string
```
